class A {
  var x = 6;
  var y = 10;

  A(z, w){
	this.x = z;
	this.y = w;
  }
  prod() {
    return this.x * this.y;
  }

  set2(a, b) {
    x = a;
    y = b;
  }
}

class B extends A {

	B(x, y){
		super(x,y);
	}
  set1(a) {
    set2(a, a);
  }

  static main () {
    var b = new B(5,5);
    return b.prod();
  }
}
